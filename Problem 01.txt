Problem: if any value is null then populate previous populated value

Solution: 

WITH GROUPED_TBLE AS (
SELECT DATE_12,DAY_OF_WEEK,INVENTORY, COUNT(INVENTORY) OVER(ORDER BY DATE_12) AS RNK
FROM inventory_log ),
FINAL_TABLE AS (
SELECT DATE_12,DAY_OF_WEEK,INVENTORY,FIRST_VALUE(INVENTORY) OVER (PARTITION BY RNK) AS NEW_INVENTORY
FROM GROUPED_TBLE
)
SELECT DATE_12,DAY_OF_WEEK,NEW_INVENTORY
FROM FINAL_TABLE;
